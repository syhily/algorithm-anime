# 实现了队列操作的类
class Queue
    Q          # 保存队列元素的数组
    head       # 指向队列开头的箭头
    tail       # 指向队列末尾的箭头

    init():
        head ← 0
        tail ← 0
  
    enqueue(x):
        Q[tail++] ← x       # 将 tail 位置赋值为 x 后，将 tail 加 1

    dequeue():
        return Q[head++]         # 返回 Q[head] 的值后，将 head 加 1

    empty():
        return head = tail       # head 和 tail 相等时返回 True

# 队列操作模拟
Queue que(6)
que.enqueue(4)
que.enqueue(8)
que.enqueue(1)
que.dequeue()
que.enqueue(7)
que.dequeue()
