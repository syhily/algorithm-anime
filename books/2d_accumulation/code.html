rects ← 矩形的列表

# 叠加矩形
for rect in rects:
    x1 = rect.左上顶点.x
    y1 = rect.左上顶点.y
    x2 = rect.右下顶点.x
    y2 = rect.右下顶点.y
    A[x1][y1]++
    A[x2][y2]++
    A[x1][y2]--
    A[x2][y1]--

# 水平方向的累积和
for x ← 1 to N-1:
    for y ← 0 to M-1:
        A[x][y] ← A[x][y] + A[x-1][y]

# 垂直方向的累积和
for y ← 1 to M-1:
    for x ← 0 to N-1:
        A[x][y] ← A[x][y] + A[x][y-1]
